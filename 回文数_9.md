# 第9题

这题在第7题整数反转的基础上来做就行，前置条件判断下就ok，其它没什么可多说的了。

```js
    var isPalindrome = function(x) {
        let temp = x
        let sum = 0
        
        if(x < 0) {
            return false
        }
        if (x > 0 && x < 10) {
            return true
        }
        
        while (temp > 0) {
            sum = sum * 10 + temp % 10
            temp = Math.floor(temp/10)
        }
        
        if (sum === x) {
            return true
        }
        
        return false
    };
```
